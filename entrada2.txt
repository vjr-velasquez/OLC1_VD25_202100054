// ===== PRUEBA VECTORES =====

// 1) Vector vacío
var v1: int[] = [];
print("v1 creado");

// 2) Vector con valores
var v2: int[] = [10, 20, 30, 40];
print("v2[0] = " + v2[0]);
print("v2[3] = " + v2[3]);

// 3) Asignación por índice
v2[1] = 99;
print("v2[1] despues = " + v2[1]);

// 4) Índice como expresión
var i: int = 2;
print("v2[i] = " + v2[i]);
v2[i] = v2[i] + 5;
print("v2[i] despues = " + v2[i]);

print("FIN PRUEBA");
